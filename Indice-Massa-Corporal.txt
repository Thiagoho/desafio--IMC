// Table de Usuários
Create table usuarios(
	idUser int auto_increment primary Key
	nome varchar(100) not null,
	email varchar(100) unique not null,
	data_nascimento date,
	sexo ENUM('M', 'F', 'OUTRO'),
	altura decimal(3, 2) not null,
	data_criacao timestamp default current_timestamp
);

// Table de Classificação padrão
 
Create table classificacoes_imc (
	idClassif int auto_increment primary key,
	categoria varchar(50) not null,
	imc_minimo decimal(5, 2) not null,
	imc_maximo decimal(5, 2) not null,
	risco_saude varchar(100) not null,
);


// Relacionamentos sugeridos

create table registro_imc (
	idRegi int auto_increment primary key,
	idUser int not null,
	peso decimal(5, 2) not null,
	imc decimal(5, 2) not null,
	idClassif int,
	data_registro timestamp default current_timestamp,
	observacoes text,
	foreign key (idUser) references usuarios(idUser),
	foreign key (idClassif) references classificacoes_imc(idClassif)
);
